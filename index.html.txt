    <script>
        /* 星空動畫邏輯 */
        const canvas = document.getElementById('starfield');
        const ctx = canvas.getContext('2d');
        let width, height;
        let stars = [];

        // 初始化畫布尺寸
        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
            initStars();
        }

        // 星星物件
        class Star {
            constructor() {
                this.x = Math.random() * width;
                this.y = Math.random() * height;
                this.size = Math.random() * 2 + 0.1;
                this.speed = Math.random() * 0.5 + 0.1;
                this.opacity = Math.random();
                this.fadeDir = Math.random() > 0.5 ? 1 : -1;
            }

            update() {
                // 星星緩慢向上移動 (模擬微小粒子)
                this.y -= this.speed;
                if (this.y < 0) {
                    this.y = height;
                    this.x = Math.random() * width;
                }

                // 閃爍效果
                this.opacity += 0.01 * this.fadeDir;
                if (this.opacity >= 1 || this.opacity <= 0.1) {
                    this.fadeDir *= -1;
                }
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(0, 243, 255, ${this.opacity})`;
                ctx.fill();
            }
        }

        function initStars() {
            stars = [];
            // 根據螢幕寬度決定星星數量
            const numStars = Math.floor((width * height) / 4000);
            for (let i = 0; i < numStars; i++) {
                stars.push(new Star());
            }
        }

        function animate() {
            ctx.clearRect(0, 0, width, height);
            stars.forEach(star => {
                star.update();
                star.draw();
            });
            requestAnimationFrame(animate);
        }

        // 綁定事件與啟動動畫
        window.addEventListener('resize', resize);
        resize();
        animate();

        /* 按鈕點擊互動模擬 */
        function startSystem(e) {
            e.preventDefault();
            const btn = e.target;
            const originalText = btn.innerText;
            
            btn.innerText = "[ 系統初始化中... 載入數據... ]";
            btn.style.color = "#bc13fe";
            btn.style.borderColor = "#bc13fe";
            btn.style.boxShadow = "0 0 20px rgba(188, 19, 254, 0.6)";
            
            // 模擬進度延遲
            setTimeout(() => {
                alert(">> 系統權限確認。\n>> 正在解壓縮台灣空汙分析模型與資料庫...\n\n(這是一個期末專題的模擬執行動作，您可以將此按鈕連結至實際的專案檔案或報告！)");
                // 恢復原本狀態
                btn.innerText = originalText;
                btn.style.color = "";
                btn.style.borderColor = "";
                btn.style.boxShadow = "";
            }, 1500);
        }
    </script>
